The target variable is Status. Code it into a binary variable:

  • If the Status contains Fully Paid, code it as 0;
  • If the Status contains Charged Off or Late or Default or In Grace Period, code it as 1;
  • Drop all records with the other statuses.

Report the distribution of the prediction target (i.e., number of 1's and 0's).

def code_status(status):
    if pd.isna(status):
        return None
    status = str(status)
    if 'Fully Paid' in status:
        return 0
    elif any(x in status for x in ['Charged Off', 'Late', 'Default', 'In Grace Period']):
        return 1
    else:
        return None

df['target'] = df['Status'].apply(code_status)


df_clean = df.dropna(subset=['target'])
df_clean['target'] = df_clean['target'].astype(int)

print(f"\nTarget Variable Distribution:")

# Build TF-IDF vectorizer 
vectorizer = TfidfVectorizer(
    max_features=5000,  # Limit features 
    min_df=5,           # Ignore terms that appear in less than x documents
    max_df=0.8,         # Ignore terms that appear in more than x % of documents
    ngram_range=(1, 2), # Include unigrams and bigrams (1,2)
    strip_accents='unicode',
    lowercase=True,
    stop_words='english'
)

# Fit vectorizer on training data only

vectorizer.fit(X_train)

# Transform both training and test sets

X_train_tfidf = vectorizer.transform(X_train)
